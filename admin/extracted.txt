        window.AuthManager = {
            REPO: 'atadorukcakmak/dinomoreWebsite',
            FILE_PATH: 'admin/admins.json',
            FALLBACK_P_HASH: 'af7aadb5ac0d2e1c8e3b1dc9d6dbb98e3ccab87b15cfe04bfe899c35cb0d30f4',
            _remoteAdmins: [],
            _fileSha: null,

            async hash(str) {
                const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(str));
                return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2, '0')).join('');
            },

            getToken() { return localStorage.getItem('dino_gh_token') || ''; },
            setToken(t) { localStorage.setItem('dino_gh_token', t); },

            async fetchRemoteAdmins() {
                try {
                    const r = await fetch(`admins.json?t=${Date.now()}`);
                    if (r.ok) this._remoteAdmins = await r.json();
                } catch(e) { console.warn('Failed to fetch admins.json', e); }
                return this._remoteAdmins;
            },

            async fetchGitHubFile() {
                const token = this.getToken();
                if (!token) return null;
                try {
                    const r = await fetch(`https://api.github.com/repos/${this.REPO}/contents/${this.FILE_PATH}`, {
                        headers: { 'Authorization': `token ${token}`, 'Accept': 'application/vnd.github.v3+json' }
                    });
                    if (r.ok) { const d = await r.json(); this._fileSha = d.sha; return JSON.parse(atob(d.content)); }
                } catch(e) { console.warn('GitHub fetch failed', e); }
                return null;
            },

            async pushToGitHub(admins, msg) {
                const token = this.getToken();
                if (!token) return { success: false, msg: 'GitHub token ayarlanmam├ä┬▒├à┼©' };
                try {
                    if (!this._fileSha) await this.fetchGitHubFile();
                    const content = btoa(unescape(encodeURIComponent(JSON.stringify(admins, null, 4))));
                    const r = await fetch(`https://api.github.com/repos/${this.REPO}/contents/${this.FILE_PATH}`, {
                        method: 'PUT',
                        headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' },
                        body: JSON.stringify({ message: msg || 'Admin update', content, sha: this._fileSha })
                    });
                    if (r.ok) { const d = await r.json(); this._fileSha = d.content.sha; return { success: true }; }
                    const err = await r.json();
                    return { success: false, msg: err.message || 'GitHub API error' };
                } catch(e) { return { success: false, msg: e.message }; }
            },

            getLocalAdmins() { return JSON.parse(localStorage.getItem('dino_admins_local') || '[]'); },
            saveLocalAdmins(a) { localStorage.setItem('dino_admins_local', JSON.stringify(a)); },

            getAllAdmins() {
                const remote = this._remoteAdmins.map(a => ({ ...a, source: 'deployed' }));
                const local = this.getLocalAdmins().map(a => ({ ...a, source: 'local' }));
                const merged = [...remote];
                local.forEach(la => { if (!merged.some(m => m.u === la.u)) merged.push(la); });
                return merged;
            },

            async validate(user, pass) {
                const u = await this.hash(user.trim().toLowerCase());
                const p = await this.hash(pass);
                await this.fetchRemoteAdmins();
                const all = this.getAllAdmins();
                const primary = this._remoteAdmins[0];
                const found = all.find(a => a.u === u && (a.p === p || (primary && a.u === primary.u && p === this.FALLBACK_P_HASH)));
                if (found) {
                    let local = this.getLocalAdmins();
                    const idx = local.findIndex(a => a.u === u);
                    if (idx !== -1) local[idx].lastLogin = new Date().toISOString();
                    else local.push({ u, name: found.name || user.trim(), lastLogin: new Date().toISOString() });
                    this.saveLocalAdmins(local);
                    return true;
                }
                return false;
            },

            async addAdmin(user, pass) {
                const u = await this.hash(user.trim().toLowerCase());
                const p = await this.hash(pass);
                const all = this.getAllAdmins();
                if (all.some(a => a.u === u)) return { success: false, msg: 'User already exists' };
                const newAdmin = { u, p, name: user.trim(), role: 'admin', created: new Date().toISOString() };
                // Yerel kay├ä┬▒t
                let local = this.getLocalAdmins();
                local.push(newAdmin);
                this.saveLocalAdmins(local);
                // GitHub'a otomatik push
                let deployed = false;
                if (this.getToken()) {
                    const remote = await this.fetchGitHubFile() || [...this._remoteAdmins];
                    remote.push({ u, p, name: user.trim(), role: 'admin', created: new Date().toISOString() });
                    const res = await this.pushToGitHub(remote, `Yeni admin eklendi: ${user.trim()}`);
                    deployed = res.success;
                    if (res.success) this._remoteAdmins = remote;
                }
                if (window.renderAdminTables) window.renderAdminTables();
                return { success: true, deployed };
            },

            async deleteAdmin(uHash) {
                const primary = this._remoteAdmins[0];
                if (primary && uHash === primary.u) return { success: false, msg: 'Ana admin silinemez' };
                let local = this.getLocalAdmins().filter(a => a.u !== uHash);
                this.saveLocalAdmins(local);
                let deployed = false;
                if (this.getToken()) {
                    const remote = await this.fetchGitHubFile() || [...this._remoteAdmins];
                    const filtered = remote.filter(a => a.u !== uHash);
                    if (filtered.length < remote.length) {
                        const res = await this.pushToGitHub(filtered, 'Admin silindi');
                        deployed = res.success;
                        if (res.success) this._remoteAdmins = filtered;
                    }
                }
                if (window.renderAdminTables) window.renderAdminTables();
                return { success: true, deployed };
            },

            async updateAdmin(oldUser, newUser, newPass) {
                const oldU = await this.hash(oldUser.trim().toLowerCase());
                const newU = await this.hash(newUser.trim().toLowerCase());
                const newP = await this.hash(newPass);
                // Update local
                let local = this.getLocalAdmins();
                const li = local.findIndex(a => a.u === oldU);
                if (li !== -1) local[li] = { ...local[li], u: newU, p: newP, name: newUser.trim() };
                else local.push({ u: newU, p: newP, name: newUser.trim(), created: new Date().toISOString() });
                this.saveLocalAdmins(local);
                // Update remote
                let deployed = false;
                if (this.getToken()) {
                    const remote = await this.fetchGitHubFile() || [...this._remoteAdmins];
                    const ri = remote.findIndex(a => a.u === oldU);
                    if (ri !== -1) { remote[ri] = { ...remote[ri], u: newU, p: newP, name: newUser.trim() }; }
                    else { remote.push({ u: newU, p: newP, name: newUser.trim(), role: 'admin', created: new Date().toISOString() }); }
                    const res = await this.pushToGitHub(remote, `Admin g├â┬╝ncellendi: ${newUser.trim()}`);
                    deployed = res.success;
                    if (res.success) this._remoteAdmins = remote;
                }
                if (window.renderAdminTables) window.renderAdminTables();
                return { success: true, deployed };
            },

            migrateOldFormat() {
                const old = JSON.parse(localStorage.getItem('dino_admins') || '[]');
                if (old.length > 0) {
                    const local = this.getLocalAdmins();
                    old.forEach(o => { if (!local.some(l => l.u === o.u)) local.push({ ...o, source: 'migrated' }); });
                    this.saveLocalAdmins(local);
                    localStorage.removeItem('dino_admins');
                }
            }
        };

        window.renderAdminTables = function() {
            const admins = AuthManager.getAllAdmins();
            const localAdmins = AuthManager.getLocalAdmins();
            const now = Date.now();
            
            // All Admins
            const allTbody = document.getElementById('allAdminsTable');
            if (allTbody) {
                allTbody.innerHTML = admins.map(a => {
                    const isDeployed = a.source === 'deployed';
                    const isPrimary = AuthManager._remoteAdmins[0] && a.u === AuthManager._remoteAdmins[0].u;
                    const sourceLabel = isDeployed 
                        ? '<span style="background:var(--accent);color:#000;padding:2px 6px;border-radius:4px;font-size:0.65rem;font-weight:700;">DEPLOY</span>'
                        : '<span style="background:var(--purple);color:#fff;padding:2px 6px;border-radius:4px;font-size:0.65rem;font-weight:700;">YEREL</span>';
                    const deleteBtn = isPrimary
                        ? '<i class="fas fa-shield-alt" style="color:var(--accent);" title="Ana admin silinemez"></i>'
                        : `<button onclick="Actions.handleDeleteAdmin('${a.u}')" style="background:none; border:none; color:var(--pink); cursor:pointer; font-size:0.85rem;" title="Sil"><i class="fas fa-trash"></i></button>`;
                    return `<tr>
                        <td><strong>${a.name || 'Admin'}</strong> ${sourceLabel}</td>
                        <td>${a.created ? new Date(a.created).toLocaleDateString() : '-'}</td>
                        <td>${deleteBtn}</td>
                    </tr>`;
                }).join('');
            }

            // Active Admins (login bilgisi localStorage'dan)
            const activeTbody = document.getElementById('activeAdminsTable');
            if (activeTbody) {
                const allWithLogin = [...admins];
                // localStorage'daki login bilgilerini ekle
                localAdmins.forEach(la => {
                    if (la.lastLogin) {
                        const existing = allWithLogin.find(a => a.u === la.u);
                        if (existing) existing.lastLogin = la.lastLogin;
                    }
                });
                const activeOnes = allWithLogin.filter(a => a.lastLogin && (now - new Date(a.lastLogin).getTime() < 86400000));
                activeTbody.innerHTML = activeOnes.map(a => `
                    <tr>
                        <td>${a.name || 'Admin'}</td>
                        <td>${new Date(a.lastLogin).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}</td>
                        <td><span class="pulse-dot"></span> <span style="color:#00ff88;">Aktif</span></td>
                    </tr>
                `).join('') || '<tr><td colspan="3" style="text-align:center; opacity:0.5; padding:10px;">Son 24 saatte giri├à┼© yok</td></tr>';
            }
        };
